 2-tar xvzf apache-cassandra-3.11.15-bin.tar.gz 
3-sudo mv apache-cassandra-3.11.15 /usr/local/cassandra
4-nano ~/.bashrc
5-export PATH=$PATH:/usr/local/cassandra/bin
6-source ~/.bashrc
